let swap = function (array, pos1, pos2) {
    var temp = array[pos1];
    array[pos1] = array[pos2];
    array[pos2] = temp;
};
let combinations = (array, output, n) => {
    n = n || array.length;
    if (n === 1) {
        output(array);
    } else {
        for (var i = 1; i <= n; i += 1) {
        combinations(array, output, n - 1);
        if (n % 2) {
            var j = 1;
        } else {
            var j = i;
        }
        swap(array, j - 1, n - 1); 
    }
}
};

a = [1, 2, 3]

def pn(start, end):
    if end==start:
        print ('out= ', a)
        return
    for i in range(start, end+1):
        #swapping
        a[i],a[start] = a[start],a[i]
        print(a, 'sw= ', start, i, 'pn= ',start+1,end,'for= ', start,end)
        #calling permutation function
        #by keeping the element at the index start fixed

        pn(start+1, end)
        #restoring the array
        a[i],a[start] = a[start],a[i]

при первом вызове ничего не меняет
